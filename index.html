function addXP(amount) { if (!userData) return; let newXp = (userData.xp || 0) + amount; let nextLevelBoundary = (userData.level || 1) * 1000; // 1000 XP per level let newLevel = userData.level || 1; if (newXp >= nextLevelBoundary) { newLevel++; newXp -= nextLevelBoundary; showToast(`Level Up! You are now Level ${newLevel} 🎉`); } saveData({ xp: newXp, level: newLevel }, { merge: true }); updateProfileUI(); } function updateStreak(didStudyHourOrHabit) { const today = new Date().toISOString().split('T')[0]; let lastUpdate = userData.lastStreakUpdate || ""; let currentStreak = userData.streak || 0; if (didStudyHourOrHabit && lastUpdate !== today) { // If yesterday was missed, and last update wasn't today, reset to 1 const yesterday = new Date(); yesterday.setDate(yesterday.getDate() - 1); const yStr = yesterday.toISOString().split('T')[0]; currentStreak = (lastUpdate === yStr) ? currentStreak + 1 : 1; saveData({ streak: currentStreak, lastStreakUpdate: today }, { merge: true }); showToast(`Streak 🔥 updated: ${currentStreak} days!`); } }
